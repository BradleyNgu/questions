$ = \sum^{100}_{k=0} \binom{100}{k} {(7x)}^{k} {(-13y)}^{100-k} $ \\
$ = \sum^{100}_{k=0} \binom{100}{20} {(7x)}^{20} {(-13y)}^{100-20} $ \\
$ = \binom{100}{20} {(7)}^{20} x^{20} {(-13)}^{80} y^{80} $ \\
$ = \binom{100}{20} {(7)}^{20} {(-13)}^{80} x^{20} y^{80} $
$ = \binom{100}{20} {(7)}^{20} {(13)}^{80} x^{20} y^{80} $
